```javascript
function generateBusinessCard(name, title) {
  const canvas = document.createElement('canvas');
  canvas.width = 1050;  // 3.5 inches at 300 DPI
  canvas.height = 600;  // 2 inches at 300 DPI
  const ctx = canvas.getContext('2d');

  // Front Side
  // Background
  ctx.fillStyle = '#ffffff';
  ctx.fillRect(0, 0, 1050, 600);

  // Gold accent bar
  ctx.fillStyle = '#d4af37';
  ctx.fillRect(0, 0, 1050, 15);

  // Logo (you'll need to load image)
  const logo = new Image();
  logo.src = 'logo.png';
  logo.onload = function() {
    ctx.drawImage(logo, 50, 80, 150, 150);
  };

  // Company Name
  ctx.fillStyle = '#1e40af';
  ctx.font = 'bold 32px Montserrat';
  ctx.fillText('QANZAK GLOBAL PROPERTIES', 220, 120);

  // Person Name
  ctx.fillStyle = '#1f2937';
  ctx.font = 'bold 42px Montserrat';
  ctx.fillText(name, 220, 200);

  // Title
  ctx.fillStyle = '#6b7280';
  ctx.font = '24px Montserrat';
  ctx.fillText(title, 220, 240);

  // Contact Info
  ctx.fillStyle = '#1f2937';
  ctx.font = '22px Arial';
  ctx.fillText('ğŸ“ +92 331 1479800', 220, 320);
  ctx.fillText('ğŸ“§ info@qanzakglobal.com', 220, 360);
  ctx.fillText('ğŸŒ properties.qanzakglobal.com', 220, 400);

  // Address
  ctx.fillStyle = '#6b7280';
  ctx.font = '18px Arial';
  ctx.fillText('House 66, F-11/1, Islamabad, Pakistan', 220, 450);

  // Download
  canvas.toBlob(blob => {
    const url = URL.createObjectURL(blob);
    const link = document.createElement('a');
    link.download = `business-card-${name.replace(' ', '-')}.png`;
    link.href = url;
    link.click();
  });
}
```

```javascript
// Using jsPDF library
function generateBrochurePDF() {
  const { jsPDF } = window.jspdf;
  const doc = new jsPDF({
    orientation: 'portrait',
    unit: 'mm',
    format: 'a4'
  });

  // Page 1 - Cover
  doc.setFillColor(30, 64, 175);
  doc.rect(0, 0, 210, 297, 'F');
  
  doc.setTextColor(255, 255, 255);
  doc.setFontSize(48);
  doc.setFont('helvetica', 'bold');
  doc.text('QANZAK GLOBAL', 105, 120, { align: 'center' });
  doc.text('PROPERTIES', 105, 140, { align: 'center' });
  
  doc.setFontSize(20);
  doc.setFont('helvetica', 'normal');
  doc.text('Your Trusted Real Estate Partner', 105, 160, { align: 'center' });

  // Page 2 - About Us
  doc.addPage();
  doc.setFillColor(255, 255, 255);
  doc.rect(0, 0, 210, 297, 'F');
  
  doc.setTextColor(30, 64, 175);
  doc.setFontSize(32);
  doc.setFont('helvetica', 'bold');
  doc.text('WHO WE ARE', 20, 30);
  
  doc.setTextColor(31, 41, 55);
  doc.setFontSize(12);
  doc.setFont('helvetica', 'normal');
  const aboutText = `Qanzak Global Properties is a premier real estate company serving Islamabad and Lahore. With over 15 years of industry experience, we have helped thousands of clients find their dream properties and make sound investment decisions.`;
  
  doc.text(aboutText, 20, 50, { maxWidth: 170, align: 'justify' });

  // Add more pages...

  // Save PDF
  doc.save('qanzak-brochure.pdf');
}
```
