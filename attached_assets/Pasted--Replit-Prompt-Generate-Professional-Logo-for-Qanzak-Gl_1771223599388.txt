### PNG Exports for Different Uses

Create PNG versions with transparency:

```javascript
// Export PNG from SVG
async function exportLogoPNG(svgElement, filename, width, height) {
  const canvas = document.createElement('canvas');
  canvas.width = width;
  canvas.height = height;
  const ctx = canvas.getContext('2d');
  
  const img = new Image();
  const svgBlob = new Blob([svgElement.outerHTML], {type: 'image/svg+xml'});
  const url = URL.createObjectURL(svgBlob);
  
  img.onload = () => {
    ctx.drawImage(img, 0, 0, width, height);
    canvas.toBlob((blob) => {
      const link = document.createElement('a');
      link.download = filename;
      link.href = URL.createObjectURL(blob);
      link.click();
    });
  };
  img.src = url;
}

// Generate different sizes
exportLogoPNG(svgElement, 'logo-2000.png', 2000, 1000); // High-res
exportLogoPNG(svgElement, 'logo-400.png', 400, 200);    // Web
exportLogoPNG(svgElement, 'logo-200.png', 200, 100);    // Small
```
